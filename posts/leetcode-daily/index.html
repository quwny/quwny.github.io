class Solution {
    public int repeatedNTimes(int[] nums) {

        int res = nums[0], n = nums.length;
        // è®¡æ•°
        int count = 1, curr = nums[0];
        for (int i = 1; i < n; ++i) {
            if (nums[i] == res) {
                return res;
            }

            // æŠ•ç¥¨ç®—æ³•
            if (curr == nums[i]) {
                ++count;
            } else {
                if (--count <= 0) {
                    curr = nums[i];
                    count = 1;
                }
            }

            if (count > 1) {
                return curr;
            }
        }

        return curr;
    }
}

class Solution {
    private int[] o = new int[26];

    public boolean isAlienSorted(String[] words, String order) {

        // ä¿å­˜å­—å…¸åº
        for (int i = 0; i < 26; ++i) {
            o[order.charAt(i) - 'a'] = i;
        }

        int n = words.length;

    }

    // æ¯”è¾ƒæ˜¯å¦ç¬¦åˆå­—å…¸åº
    private boolean order(String s1, String s2) {
        int len1 = s1.length();
        int len2 = s2.length();
        int j = 0, k = 0;

        while (j < len1 && k < len2) {
            char ch1 = s1.charAt(j++);
            char ch2 = s2.charAt(k++);
            if (ch1 == ch2) {
                continue;
            }
            // ä¸ç¬¦åˆå­—å…¸åº
            if (o[ch1 - 'a'] > o[ch2 - 'a']) {
                return false;

            return true;

    }
}

class Solution {

        double area = 0; // é¢ç§¯

        int minVal = Integer.MAX_VALUE, maxVal = Integer.MIN_VALUE;

class Solution {
    public Node construct(int[][] grid) {
        return dfs(grid, 0, grid.length - 1, 0, grid[0].length - 1);
    }

    private Node dfs(int[][] grid, int r0, int r1, int c0, int c1) {
        // åˆ¤æ–­å½“å‰ç½‘æ ¼çš„å€¼æ˜¯å¦éƒ½ç›¸åŒ
        boolean same = true;
        loop: for (int i = r0; i <= r1; ++i) {
            for (int j = c0; j <= c1; ++j) {
                // å¦‚æœä¸æ˜¯å…¨éƒ¨ç›¸åŒï¼Œè·³å‡ºå¾ªç¯
                if (grid[r0][c0] != grid[i][j]) {
                    same = false;
                    break loop;
                }
            }
        }
        // å¦‚æœå½“å‰ç½‘æ ¼æ•°å€¼å…¨éƒ¨ç›¸åŒï¼Œä¸ºå¶å­ç»“ç‚¹
        if (same) {
            // grid[r0][c0] == 1 -- å®˜æ–¹å†™æ³•
            return new Node(grid[r0][c0] == 1 ? true : false, true);
        }

        // åˆ’åˆ†
        Node node = new Node(true, false);
        node.topLeft = dfs(grid, r0, (r0 + r1) / 2, c0, (c0 + c1) / 2);
        node.topRight = dfs(grid, r0, (r0 + r1) / 2, (c0 + c1) / 2 + 1, c1);
        node.bottomLeft = dfs(grid, (r0 + r1) / 2 + 1, r1, c0, (c0 + c1) / 2);
        node.bottomRight = dfs(grid, (r0 + r1) / 2 + 1, r1, (c0 + c1) / 2 + 1, c1);
        return node;
    }

// è§£é¢˜æ€è·¯ï¼šä½¿ç”¨åŒæŒ‡é’ˆï¼Œåˆ†åˆ«ä»å¤´å°¾è¿›è¡Œéå†ï¼Œå¤´æŒ‡é’ˆé‡åˆ°å¶æ•°è·³è¿‡ï¼Œå°¾æŒ‡é’ˆé‡åˆ°å¥‡æ•°è·³è¿‡ï¼Œå¦åˆ™è¿›è¡Œäº¤æ¢ä¸¤æ•°ã€‚
class Solution {
    public int[] sortArrayByParity(int[] nums) {
        int n = nums.length;
        int left = 0, right = n - 1;
        while (left < right) {
            // å¤´æŒ‡é’ˆéå†
            while (left < right && nums[left] % 2 == 0)
                ++left;
            // å°¾æŒ‡é’ˆéå†
            while (right > left && nums[right] % 2 == 1)
                --right;
            // äº¤æ¢ä¸¤ä¸ªæ•°
            int temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
        }
        return nums;
    }
}
