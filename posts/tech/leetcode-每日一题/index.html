        /**
         * è§£é¢˜æ€è·¯ï¼šéå†å­—ç¬¦ä¸²ï¼ŒåŒæŒ‡é’ˆåˆ†åˆ«å¤´å°¾éå† 0~n+1ï¼Œå¦‚æœæ˜¯ 'I'ï¼Œå¤´æŒ‡é’ˆ + 1ï¼Œå¦åˆ™å°¾æŒ‡é’ˆ - 1ã€‚

class Solution {
    public int findTheWinner(int n, int k) {
        /**
         * è§£é¢˜æ€è·¯ï¼š
         * æŒ‰ç…§æ¸¸æˆè§„åˆ™è¿›è¡Œæ¨¡æ‹Ÿï¼Œä½¿ç”¨é˜Ÿåˆ—ï¼Œä»å½“å‰ curr éå† k - 1 ä¸ªæ•°ï¼Œå¹¶æ·»åŠ åˆ°é˜Ÿå°¾ï¼Œç¬¬ k ä¸ªæ•°å‡ºé˜Ÿã€‚

        Queue<Integer> queue = new ArrayDeque<>();
        // æ·»åŠ  n ä¸ªå…ƒç´ 
        for (int i = 1; i <= n; ++i) {
            queue.offer(i);
        }

        while (n-- > 1) {
            for (int i = 1; i < k; ++i) {
                queue.offer(queue.poll());
            }
            queue.poll();
        }
        return queue.peek();
    }
}

    private Node dfs(int[][] grid, int r0, int r1, int c0, int c1) {
        // åˆ¤æ–­å½“å‰ç½‘æ ¼çš„å€¼æ˜¯å¦éƒ½ç›¸åŒ

class Solution {
    public int[] sortArrayByParity(int[] nums) {
        int n = nums.length;
        int left = 0, right = n - 1;
        while (left < right) {
            // å¤´æŒ‡é’ˆéå†
            while (left < right && nums[left] % 2 == 0)
                ++left;
            // å°¾æŒ‡é’ˆéå†
            while (right > left && nums[right] % 2 == 1)
                --right;
            // äº¤æ¢ä¸¤ä¸ªæ•°
            int temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
        }
        return nums;
    }
}
